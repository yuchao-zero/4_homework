1. 编写一个方法不使用 * 完成两个数相乘
```
class Multiply{
	public static void main(String[] args){
		
		 System.out.println(multiply(4 , 8));
	} 
	//需要采用二进制位移方法来做
	static int multiply(int n1, int n2){
		int sum = 0;
		while (n1!=0) {

			if (n1%2!=0) { 	      //当n1为奇数的时候，将n2的值加给sum
				sum = sum + n2;
			}
			n1 = n1 >> 1;      // n1 右移   除以2
			n2 = n2 << 1;      // n2 左移   乘以2
		}
	
		return sum;
	}
}
```

```
class Test{
	//测试函数
	public static void main(String[] args){
		array(new int[]{1,2,3,3,4,4,5});

	} 
	static int[] array(int []arr){
		int a=arr.length;
		for(int i=0;i<a;i++){
			for(int j=i+1;j<a;j++){
				if(arr[i]==arr[j]){
					arr[j]=arr[a-1];
					a--;
					j--;
				}
			}
		}
		for (int one : arr) {
            System.out.printf("%d ", one);
        }

        System.out.println();

		int[] art = new int[a];
		for(int i=0;i<a;i++){
			art[i] = arr[i];
		}
		for (int one : art) {
            System.out.printf("%d ", one);
        }
        System.out.println();
        return art;
	}
}
```
